merge-meshes

Merges multiple meshes into one

Install

$ npm install merge-meshes

Usage

var mergeMeshes = require('merge-meshes');

var meshA = {
  cells: [
    [0, 1, 2]
  ],
  positions: [
    [ 0, 0, 0],
    [ 1, 0, 0],
    [ 0, 1, 0],
  ]
}

var meshB = {
  cells: [
    [0, 1, 2],
  ],
  positions: [
    [ 0, 0, 0],
    [-1, 0, 0],
    [ 0, 1, 0]
  ]
}

var mesh = mergeMeshes([meshA, meshB]);

console.log(mesh);
